}
 })();

adds one to the counter (as above). The count.reset( ) method sets the counter to 0.

var add = (function () {
            var counter = 0;
            function reset() {
                counter = 0;
                return 0;
            }
            function add() {

            }
            var count = {
                add: add,

            };
